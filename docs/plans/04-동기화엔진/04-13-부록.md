## 부록 A: 전체 설정 인터페이스

```typescript
/**
 * 동기화 엔진 전체 설정
 * 모든 하위 모듈의 설정을 통합한 최상위 인터페이스
 */
interface SyncEngineConfig {
  // === 감지 전략 설정 ===
  detection: {
    primary: {
      enabled: true;
      intervalMs: 5000;
      maxEventsPerPoll: 100;
      timeoutMs: 10000;
    };
    secondary: {
      enabled: true;
      intervalMs: 600000;          // 10분
      concurrentFolderScans: 3;
      timeoutMs: 120000;           // 2분
    };
    tertiary: {
      enabled: true;
      intervalMs: 3600000;         // 1시간
      batchSize: 50;
      maxFileAge: 2592000000;      // 30일
    };
  };

  // === 이벤트 처리 설정 ===
  events: {
    filter: FilterConfig;
    batch: BatchConfig;
    deduplication: {
      cacheSize: 10000;
      cacheTtlMs: 86400000;        // 24시간
    };
  };

  // === 다운로드 설정 ===
  download: DownloadManagerConfig;

  // === 업로드 설정 ===
  upload: UploadQueueConfig;

  // === 재시도 설정 ===
  retry: {
    network: NetworkRetryPolicy;
    auth: AuthRetryPolicy;
    rateLimit: RateLimitRetryPolicy;
    fileSystem: FileSystemRetryPolicy;
    server: ServerRetryPolicy;
    dlq: DLQConfig;
  };

  // === 서킷 브레이커 설정 ===
  circuitBreaker: CircuitBreakerConfig;

  // === 체크포인트 설정 ===
  checkpoint: {
    flushIntervalMs: 30000;
    immediateKeys: string[];       // 즉시 저장할 키 목록
  };

  // === 인증 설정 ===
  auth: {
    playwright: PlaywrightLoginConfig;
    cookieStorePath: string;
    sessionCheckIntervalMs: 60000;
    maxReauthAttempts: 3;
  };

  // === HTTP 클라이언트 설정 ===
  http: HttpClientConfig;

  // === 디렉토리 설정 ===
  directories: DirectoryConfig;

  // === DB 설정 ===
  database: DBConfig & { path: string };

  // === 로깅 설정 ===
  logging: {
    level: 'debug' | 'info' | 'warn' | 'error';
    file: string;
    maxFileSize: number;
    maxFiles: number;
  };
}
```
